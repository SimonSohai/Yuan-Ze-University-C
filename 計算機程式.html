Coursera自學的測驗

(這題主要是考收到的一些日期，並要計算出該日期為星期幾)我先寫了一個判斷是來判段該年是否為閏年，再計算每個輸入行，若輸入的「月」有誤請輸出 -1若輸入的「月」無誤但「日」有誤請輸出 -2，再定義每個月份的日期，最後就可推斷出該日期為星期幾。



程式碼

#include
int main ()
{
int y, d;
int n;
int month, day;
int leap_year;
int days; scanf ( "%d%d", &y, &d );
scanf ( "%d", &n );

leap_year = y % 400 == 0 || ( y % 100 && y % 4 == 0 ); //閏年還是非閏年

for ( int i = 0; i < n; i++ ) { //計算每個輸入行
if ( i ) printf ( " " );

days = d;
scanf ( "%d%d", &month, &day );

if ( month < 1 || month > 12 ) { //若輸入的「月」有誤請輸出 -1
printf ( "-1\t" );
continue;
}
if ( day < 1 ) { //若輸入的「月」無誤但「日」有誤請輸出 -2
printf ( "-2\t" ); continue;
}
else {
switch ( month ) {
case 2:
if ( ( leap_year && day > 29 ) ||
( !leap_year && day > 28 ) ) {
printf ( "-2\t" );
continue;
}
break;
case 4: case 6: case 9: case 11:
if ( day > 30 ) {
printf ( "-2\t" );
continue;
}
break;
default:
if ( day > 31 ) {
printf ( "-2\t" );
continue;
}
}
}
for ( int m = 1; m < month; m++ ) {
switch ( m ) {

case 2: //定義2月份的日期
if ( leap_year ) days += 29;
else days += 28;
break;
case 4: case 6: case 9: case 11: //定義為30天的月份
days += 30;
break;
default: //其餘的月份定義為31天
>days += 31;
}
}
days += ( day - 1 );
printf ( "%d\t", days % 7 );
}
return 0;
}
